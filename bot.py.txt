import logging
import os
import sqlite3
import re
from datetime import datetime
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes

# Configuration du logging
logging.basicConfig(
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    level=logging.INFO
)

# Récupérer le token depuis les variables d'environnement
TOKEN = os.environ.get('TELEGRAM_TOKEN')

if not TOKEN:
    raise ValueError("TELEGRAM_TOKEN non défini dans les variables d'environnement")

# ============ BASE DE DONNÉES ============
DB_PATH = 'finaansoovy_bot.db'

def get_connection():
    return sqlite3.connect(DB_PATH)

def init_db():
    conn = get_connection()
    cursor = conn.cursor()
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            telegram_id INTEGER UNIQUE,
            username TEXT,
            created_at TIMESTAMP
        )
    ''')
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS transactions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id INTEGER,
            amount REAL,
            category TEXT,
            description TEXT,
            type TEXT,
            date TIMESTAMP,
            FOREIGN KEY (user_id) REFERENCES users (id)
        )
    ''')
    
    conn.commit()
    conn.close()
    logging.info("✅ Base de données initialisée")

def add_user(telegram_id, username):
    conn = get_connection()
    cursor = conn.cursor()
    
    try:
        cursor.execute('''
            INSERT INTO users (telegram_id, username, created_at)
            VALUES (?, ?, ?)
        ''', (telegram_id, username, datetime.now()))
        conn.commit()
        user_id = cursor.lastrowid
    except sqlite3.IntegrityError:
        cursor.execute('SELECT id FROM users WHERE telegram_id = ?', (telegram_id,))
        user_id = cursor.fetchone()[0]
    finally:
        conn.close()
    return user_id

def add_transaction(user_id, amount, category, description, trans_type):
    conn = get_connection()
    cursor = conn.cursor()
    
    cursor.execute('''
        INSERT INTO transactions (user_id, amount, category, description, type, date)
        VALUES (?, ?, ?, ?, ?, ?)
    ''', (user_id, amount, category, description, trans_type, datetime.now()))
    
    conn.commit()
    conn.close()

def get_user(telegram_id):
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM users WHERE telegram_id = ?', (telegram_id,))
    user = cursor.fetchone()
    conn.close()
    return user

def get_stats(user_id):
    conn = get_connection()
    cursor = conn.cursor()
    
    cursor.execute('''
        SELECT SUM(amount) FROM transactions 
        WHERE user_id = ? AND type = 'expense'
    ''', (user_id,))
    total_expenses = cursor.fetchone()[0] or 0
    
    cursor.execute('''
        SELECT SUM(amount) FROM transactions 
        WHERE user_id = ? AND type = 'income'
    ''', (user_id,))
    total_income = cursor.fetchone()[0] or 0
    
    conn.close()
    return total_income, total_expenses

# ============ HANDLERS TELEGRAM ============

    user_id = update.effective_user.id
    username = update.effective_user.username
    
    db_user_id = add_user(user_id, username)
    
    await update.message.reply_text(
        "👋 Привет! Я @finaansoovy_bot - ваш финансовый помощник.\n\n"
        "Я работаю 24/7 и готов записывать ваши расходы!\n\n"
        "📝 *Попробуйте:*\n"
        "• 'расход 500₽ продукты'\n"
        "• 'зарплата 50000₽'\n\n"
        "📊 *Команды:*\n"
        "/stats - Моя статистика\n"
        "/help - Помощь",
        parse_mode='Markdown'
    )

async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "📚 *Помощь @finaansoovy_bot*\n\n"
        "*Команды:*\n"
        "/start - Запустить бота\n"
        "/help - Эта справка\n"
        "/stats - Статистика\n\n"
        "*Как записывать расходы:*\n"
        "расход 500₽ продукты\n"
        "потратил 300₽ на транспорт\n"
        "купил 1500₽ одежда\n\n"
        "*Как записывать доходы:*\n"
        "зарплата 50000₽\n"
        "доход 10000₽ фриланс\n\n"
        "*Примеры:*\n"
        "расход 250₽ ресторан\n"
        "зарплата 45000₽",
        parse_mode='Markdown'
    )

async def stats(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    user = get_user(user_id)
    
    if not user:
        await update.message.reply_text("Используйте /start для начала")
        return
    
    total_income, total_expenses = get_stats(user[0])
    savings = total_income - total_expenses
    
    await update.message.reply_text(
        f"📊 *Статистика @finaansoovy_bot*\n\n"
        f"💰 *Доходы:* {total_income:.2f}₽\n"
        f"💸 *Расходы:* {total_expenses:.2f}₽\n"
        f"💪 *Накопления:* {savings:.2f}₽\n\n"
        f"Продолжайте в том же духе! 🎉",
        parse_mode='Markdown'
    )

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    message = update.message.text.lower()
    
    user = get_user(user_id)
    if not user:
        await update.message.reply_text("Используйте /start для начала")
        return
    
    # Rechercher une somme (en roubles)
    amount_match = re.search(r'(\d+(?:[.,]\d+)?)\s*[₽руб]', message)
    if not amount_match:
        await update.message.reply_text(
            "❓ Я не нашел сумму.\n\n"
            "📝 *Пример:*\n"
            "расход 500₽ продукты",
            parse_mode='Markdown'
        )
        return
    
    amount_str = amount_match.group(1).replace(',', '.')
    amount = float(amount_str)
    
    # Déterminer si c'est une dépense ou un revenu
    expense_keywords = ['расход', 'потратил', 'купил', 'оплатил', 'заплатил']
    income_keywords = ['зарплата', 'доход', 'заработал', 'получил', 'премия']
    
    if any(word in message for word in expense_keywords):
        trans_type = 'expense'
        category = "Разное"
        
        categories = {
            'продукты': 'Продукты',
            'еда': 'Продукты',
            'ресторан': 'Ресторан',
            'кафе': 'Ресторан',
            'транспорт': 'Транспорт',
            'такси': 'Транспорт',
            'развлечения': 'Развлечения',
            'кино': 'Развлечения',
            'одежда': 'Одежда',
            'коммунальные': 'Коммунальные',
            'квартплата': 'Коммунальные'
        }
        
        for key, cat in categories.items():
            if key in message:
                category = cat
                break
        
        response = f"✅ *Расход добавлен:* {amount}₽\n📂 *Категория:* {category}"
    
    elif any(word in message for word in income_keywords):
        trans_type = 'income'
        category = "Доход"
        
        if 'зарплата' in message:
            category = "Зарплата"
        elif 'премия' in message:
            category = "Премия"
        elif 'фриланс' in message:
            category = "Фриланс"
        
        response = f"✅ *Доход добавлен:* {amount}₽\n📂 *Категория:* {category}"
    
    else:
        await update.message.reply_text(
            "❓ Я не понял, это расход или доход.\n\n"
            "Используйте ключевые слова:\n"
            "• расход, потратил, купил\n"
            "• зарплата, доход, заработал\n\n"
            "📝 *Пример:*\n"
            "расход 500₽ продукты",
            parse_mode='Markdown'
        )
        return
    
    add_transaction(user[0], amount, category, update.message.text, trans_type)
    
    # Ajouter un petit conseil
    response += "\n\n💡 *Совет:* Регулярно записывайте расходы для лучшего контроля бюджета!"
    
    await update.message.reply_text(response, parse_mode='Markdown')

# ============ MAIN ============

def main():
    # Initialiser la base de données
    init_db()
    logging.info("🚀 Démarrage du bot @finaansoovy_bot...")
    
    # Créer l'application
    app = Application.builder().token(TOKEN).build()
    
    # Ajouter les handlers
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(CommandHandler("stats", stats))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    
    logging.info("✅ Bot @finaansoovy_bot démarré avec succès!")
    
    # Démarrer le bot
    app.run_polling()

if __name__ == "__main__":
    main()
